Que1. Consider the C program shown below:
  #include< stdio.h>
  #define print(x) printf("%d ", x)
  int x;
  void Q(int z)
  {
      z += x;
      print(z);
  }
  void P(int *y)
  {
      int x = *y + 2;
      Q(x);
      *y = x - 1;
      print(x);
  }
  int main()
  {
      x = 5;
      P(&x);
      print(x);
      return 0;
  }
Ans: a. 12 7 6
        b. 22 12 11
        c. 14 6 6
        d. 7 6 6

    • a) 12 7 6

      int a[]={1,2,3,4,5,6,7};
      int i;
      void *p=a;
      p=p+1;
      *p=23;

      cout<<*(a+i)<< " ";

  }
